Style Guide Code Block Example

Use this block when documenting a new style guide module or example.

          
<!-- [NAME] -->
<section id="name" class="block  sg-section">
  <header class="sg-section__header">
    <h2 class="sg-section__title">Name of the main section - EX: Typography</h2>
    <h3 class="sg-section__sub-title">Brief sub-title of the section - EX: Sitewide typographic elements</h3>
  </header>

  <article class="block">
    <p>Only use the article tag if the main section block documentation exceeds a simple paragraph. If used, this block
    should be used to explain all the block examples that follow.</p>
    <ol>
      <li>More than a paragraph</li>
    </ol>
  </article>

  <div id="[CHILD-NAME]-example" class="block  sg-section">
    <h4 class="sg-section__ex-title">An optional child example heading of the main section - EX: Headings</h3>
    <p>Short documentation for example don't use article tag</p>
    <div class="sg-ex-block">
      <h5 class="sg-ex-block__title">Code Example</h5>
      <!-- [CODE EXAMPLE BLOCK - USE SAME FORMATTING] -->
      <div class="sg-ex-block__sample">
      <!-- [VISUAL EXAMPLE] -->
      </div>
    </div>
  </div><!-- #[CHILD-NAME]-example -->

  <div id="[OTHER-NAME]-example" class="block  sg-section">
    <h4 class="sg-section__ex-title">An optional child example heading of the main section - EX: Headings</h3>
    <p>Short documentation for example don't use article tag</p>
    <div class="sg-ex-block">
      <h5 class="sg-ex-block__title">Code Example</h5>
      <!-- [CODE EXAMPLE BLOCK - USE SAME FORMATTING] -->
      <div class="sg-ex-block__sample">
      <!-- [VISUAL EXAMPLE] -->
      </div>
    </div>
  </div><!-- #[OTHER-NAME]-example -->
</section>
<!-- END [NAME] -->
